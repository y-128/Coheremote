// === Shell Helper ===
func shell(_ path: String, _ args: [String]) -> (Int32, String) {
    let proc = Process()
    let pipe = Pipe()
    proc.executableURL = URL(fileURLWithPath: path)
    proc.arguments = args
    proc.standardOutput = pipe
    proc.standardError = pipe
    var env = ProcessInfo.processInfo.environment
    env["LANG"] = "ja_JP.UTF-8"
    proc.environment = env
    do { try proc.run() } catch { return (-1, "launch error: \(error)") }
    let data = pipe.fileHandleForReading.readDataToEndOfFile()
    proc.waitUntilExit()
    return (proc.terminationStatus, String(data: data, encoding: .utf8) ?? "")
}

// === Network Helpers ===
func sendWoLPacket(mac: String) -> Bool {
    let cleaned = mac.replacingOccurrences(of: "-", with: ":").trimmingCharacters(in: .whitespaces)
    let parts = cleaned.split(separator: ":")
    guard parts.count == 6 else { return false }
    var macBytes: [UInt8] = []
    for part in parts {
        guard let byte = UInt8(part, radix: 16) else { return false }
        macBytes.append(byte)
    }
    var packet = Data(repeating: 0xFF, count: 6)
    for _ in 0..<16 { packet.append(contentsOf: macBytes) }

    let semaphore = DispatchSemaphore(value: 0)
    var success = false
    let connection = NWConnection(host: .ipv4(.broadcast), port: 9, using: .udp)
    connection.stateUpdateHandler = { state in
        if case .ready = state {
            connection.send(content: packet, completion: .contentProcessed { error in
                success = (error == nil)
                connection.cancel()
                semaphore.signal()
            })
        } else if case .failed = state {
            connection.cancel()
            semaphore.signal()
        }
    }
    connection.start(queue: .global())
    _ = semaphore.wait(timeout: .now() + 5)
    return success
}

func pingHost(_ host: String, timeout: Int = 3) -> Bool {
    let (code, _) = shell("/sbin/ping", ["-c", "1", "-W", "\(timeout * 1000)", host])
    return code == 0
}

func waitForHost(_ host: String, timeout: TimeInterval = 120) -> Bool {
    let deadline = Date().addingTimeInterval(timeout)
    while Date() < deadline {
        if pingHost(host) { return true }
        Thread.sleep(forTimeInterval: 3)
    }
    return false
}

// === Hover Button ===
class HoverButton: NSButton {
    override func updateTrackingAreas() {
        super.updateTrackingAreas()
        for area in trackingAreas { removeTrackingArea(area) }
        addTrackingArea(NSTrackingArea(
            rect: bounds,
            options: [.mouseEnteredAndExited, .activeAlways, .inVisibleRect],
            owner: self, userInfo: nil
        ))
    }
    override func mouseEntered(with event: NSEvent) {
        layer?.backgroundColor = NSColor.labelColor.withAlphaComponent(0.08).cgColor
        if let delegate = NSApp.delegate as? WrapperApp {
            delegate.showFooterHover(toolTip ?? "")
        }
    }
    override func mouseExited(with event: NSEvent) {
        layer?.backgroundColor = nil
        if let delegate = NSApp.delegate as? WrapperApp {
            delegate.hideFooterHover()
        }
    }
}

    var lockFilePath: String = ""

    func vmxHash() -> String {
        var hasher = Hasher()
        hasher.combine(VMX_PATH)
        let h = hasher.finalize()
        return String(format: "%08x", abs(h))
    }

    func lockDir() -> URL {
        FileManager.default.homeDirectoryForCurrentUser
            .appendingPathComponent("Library/Application Support/Coheremote/running")
    }

    func registerRunning() {
        let dir = lockDir()
        try? FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
        let pid = ProcessInfo.processInfo.processIdentifier
        let file = dir.appendingPathComponent("\(vmxHash())_\(pid)")
        lockFilePath = file.path
        FileManager.default.createFile(atPath: file.path, contents: VMX_PATH.data(using: .utf8))
    }

    func unregisterRunning() {
        if !lockFilePath.isEmpty {
            try? FileManager.default.removeItem(atPath: lockFilePath)
        }
    }

    func otherWrappersUsingVM() -> Bool {
        let dir = lockDir()
        let prefix = vmxHash() + "_"
        let myPid = "\(ProcessInfo.processInfo.processIdentifier)"
        guard let files = try? FileManager.default.contentsOfDirectory(atPath: dir.path) else {
            return false
        }
        for file in files where file.hasPrefix(prefix) {
            let pidStr = String(file.dropFirst(prefix.count))
            if pidStr == myPid { continue }
            // Check if the process is still alive
            if let pid = Int32(pidStr), kill(pid, 0) == 0 {
                return true
            } else {
                // Stale lock file, clean up
                try? FileManager.default.removeItem(at: dir.appendingPathComponent(file))
            }
        }
        return false
    }

    // MARK: - Logging

    func setupLog() {
        let logDir = FileManager.default.homeDirectoryForCurrentUser
            .appendingPathComponent("Library/Logs/Coheremote")
        try? FileManager.default.createDirectory(at: logDir, withIntermediateDirectories: true)
        let logPath = logDir.appendingPathComponent("\(LOG_NAME).log")

        if let attr = try? FileManager.default.attributesOfItem(atPath: logPath.path),
           let size = attr[.size] as? Int, size > 1_048_576 {
            let oldPath = logDir.appendingPathComponent("\(LOG_NAME).log.old")
            try? FileManager.default.removeItem(at: oldPath)
            try? FileManager.default.moveItem(at: logPath, to: oldPath)
        }

        FileManager.default.createFile(atPath: logPath.path, contents: nil)
        logHandle = FileHandle(forWritingAtPath: logPath.path)
        logHandle?.seekToEndOfFile()
    }

    func log(_ msg: String) {
        let f = DateFormatter()
        f.dateFormat = "yyyy-MM-dd HH:mm:ss"
        let line = "[\(f.string(from: Date()))] \(msg)\n"
        logHandle?.write(line.data(using: .utf8) ?? Data())
    }

    // MARK: - Keychain

    func loadPassword() {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: KEYCHAIN_SERVICE,
            kSecAttrAccount as String: "vmEncryption",
            kSecReturnData as String: true
        ]
        var item: CFTypeRef?
        if SecItemCopyMatching(query as CFDictionary, &item) == errSecSuccess,
           let data = item as? Data,
           let pw = String(data: data, encoding: .utf8) {
            encPassword = pw
        }
    }

    // MARK: - Guest Credentials

    func loadGuestPassword() {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: KEYCHAIN_SERVICE,
            kSecAttrAccount as String: "windowsPassword",
            kSecReturnData as String: true
        ]
        var item: CFTypeRef?
        if SecItemCopyMatching(query as CFDictionary, &item) == errSecSuccess,
           let data = item as? Data,
           let pw = String(data: data, encoding: .utf8) {
            winPassword = pw
            log("[KEYCHAIN] Windows password loaded")
        }
    }

    // MARK: - vmrun
    // NOTE: vmrun does not support passing passwords via stdin or file.
    // Passwords appear in process arguments and may be visible to other
    // processes running under the same user via `ps`. This is a known
    // limitation of the vmrun CLI.

    func vmrun(_ cmd: String, _ extraArgs: [String] = []) -> (Int32, String) {
        var args = ["-T", "fusion"]
        if !encPassword.isEmpty { args += ["-vp", encPassword] }
        args.append(cmd)
        args += extraArgs
        return shell(VMRUN_PATH, args)
    }

    func vmrunGuest(_ cmd: String, _ extraArgs: [String] = []) -> (Int32, String) {
        var args = ["-T", "fusion"]
        if !encPassword.isEmpty { args += ["-vp", encPassword] }
        args += ["-gu", WINDOWS_USERNAME, "-gp", winPassword]
        args.append(cmd)
        args += extraArgs
        return shell(VMRUN_PATH, args)
    }

    // MARK: - Windows App List

    func loadInstalledApps() {
        guard !winPassword.isEmpty, !appsLoading else { return }
        appsLoading = true
        log("[APPS] Loading installed apps...")

        DispatchQueue.global().async {
            var apps: [(name: String, path: String)] = []
            let basePaths = [
                "C:\\ProgramData\\Microsoft\\Windows\\Start Menu\\Programs",
                "C:\\Users\\\(WINDOWS_USERNAME)\\AppData\\Roaming\\Microsoft\\Windows\\Start Menu\\Programs"
            ]
            for basePath in basePaths {
                self.collectApps(from: basePath, into: &apps, depth: 0)
            }

            // Sort and deduplicate
            apps.sort { $0.name.lowercased() < $1.name.lowercased() }
            var seen = Set<String>()
            apps = apps.filter { seen.insert($0.name).inserted }

            DispatchQueue.main.async {
                self.cachedApps = apps
                self.appsLoading = false
                self.log("[APPS] Loaded \(apps.count) apps")
            }
        }
    }

    func collectApps(from dirPath: String, into apps: inout [(name: String, path: String)], depth: Int) {
        guard depth < 3 else { return }
        let (code, output) = vmrunGuest("listDirectoryInGuest", [VMX_PATH, dirPath])
        guard code == 0 else { return }

        let lines = output.components(separatedBy: "\n")
        for line in lines {
            let name = line.trimmingCharacters(in: .whitespacesAndNewlines)
            if name.isEmpty || name.hasPrefix("Directory list:") { continue }

            let fullPath = dirPath + "\\" + name

            if name.lowercased().hasSuffix(".lnk") {
                let displayName = String(name.dropLast(4))
                // Skip uninstallers and desktop.ini
                let lower = displayName.lowercased()
                if lower.contains("uninstall") || lower == "desktop" { continue }
                apps.append((name: displayName, path: fullPath))
            } else if name.lowercased().hasSuffix(".url") ||
                      name.lowercased().hasSuffix(".ini") ||
                      name.lowercased().hasSuffix(".txt") {
                continue
            } else {
                collectApps(from: fullPath, into: &apps, depth: depth + 1)
            }
        }
    }

    func escapeCmdArg(_ s: String) -> String {
        var r = s
        for ch in ["&", "|", "(", ")", "<", ">", "^", "%"] {
            r = r.replacingOccurrences(of: ch, with: "^\(ch)")
        }
        r = r.replacingOccurrences(of: "\"", with: "\"\"")
        return r
    }

    func launchWindowsApp(path: String) {
        log("[APP] Launching: \(path)")
        let safePath = escapeCmdArg(path)
        DispatchQueue.global().async {
            let (code, output) = self.vmrunGuest(
                "runProgramInGuest",
                [VMX_PATH, "-noWait", "-interactive", "-activeWindow",
                 "C:\\Windows\\System32\\cmd.exe",
                 "/c start \"\" \"\(safePath)\""]
            )
            self.log("[APP] exit=\(code) output=\(output)")
        }
    }

    func vmIsRunning() -> Bool {
        shell(VMRUN_PATH, ["list"]).1.contains(VMX_PATH)
    }

    // MARK: - VM Workflow

    func ensureFusion() {
        let running = NSWorkspace.shared.runningApplications.contains {
            $0.bundleIdentifier == "com.vmware.fusion"
        }
        if !running {
            status(L("Launching VMware Fusion...", "VMware Fusionを起動中..."))
            let _ = shell("/usr/bin/open", ["-g", "-a", "VMware Fusion"])
            for _ in 0..<15 {
                if shell(VMRUN_PATH, ["list"]).0 == 0 { break }
                Thread.sleep(forTimeInterval: 1)
            }
        }
    }

    func startVM() {
        if vmIsRunning() {
            log("[VM] Already running")
            return
        }
        status(L("Starting VM...", "VMを起動中..."))
        log("[VM] Opening VM with VMware Fusion")
        let _ = shell("/usr/bin/open", ["-a", "VMware Fusion", VMX_PATH])
    }

    func waitForVM() -> Bool {
        let timeout: TimeInterval = 120
        let start = Date()
        while Date().timeIntervalSince(start) < timeout {
            if vmIsRunning() {
                log("[VM] VM detected as running")
                Thread.sleep(forTimeInterval: 5)
                return true
            }
            Thread.sleep(forTimeInterval: 1)
        }
        log("[ERROR] VM start timed out")
        return false
    }

    func makeFooterButton(title: String, symbolName: String, action: Selector, frame: NSRect) -> NSButton {
        let btn = HoverButton(frame: frame)
        btn.bezelStyle = .recessed
        btn.isBordered = false
        btn.target = self
        btn.action = action
        btn.font = NSFont.systemFont(ofSize: 12)

        if let img = NSImage(systemSymbolName: symbolName, accessibilityDescription: title) {
            img.size = NSSize(width: 14, height: 14)
            btn.image = img
            btn.imagePosition = title.isEmpty ? .imageOnly : .imageLeading
        }
        if !title.isEmpty {
            btn.title = title
        }

        btn.wantsLayer = true
        return btn
    }

    func rebuildAppList() {
        guard let stack = appListStack else { return }
        stack.arrangedSubviews.forEach { $0.removeFromSuperview() }

        if appsLoading {
            let loadingLabel = NSTextField(labelWithString: L("Loading apps...", "アプリを読み込み中..."))
            loadingLabel.font = NSFont.systemFont(ofSize: 12)
            loadingLabel.textColor = .secondaryLabelColor
            loadingLabel.alignment = .center
            loadingLabel.translatesAutoresizingMaskIntoConstraints = false
            stack.addArrangedSubview(loadingLabel)
            loadingLabel.widthAnchor.constraint(equalTo: stack.widthAnchor).isActive = true
            loadingLabel.heightAnchor.constraint(equalToConstant: 36).isActive = true
            return
        }

        if cachedApps.isEmpty && connected {
            let emptyLabel = NSTextField(labelWithString: L("No apps found", "アプリが見つかりません"))
            emptyLabel.font = NSFont.systemFont(ofSize: 12)
            emptyLabel.textColor = .secondaryLabelColor
            emptyLabel.alignment = .center
            emptyLabel.translatesAutoresizingMaskIntoConstraints = false
            stack.addArrangedSubview(emptyLabel)
            emptyLabel.widthAnchor.constraint(equalTo: stack.widthAnchor).isActive = true
            emptyLabel.heightAnchor.constraint(equalToConstant: 36).isActive = true
            return
        }

        // App list
        for (index, app) in cachedApps.enumerated() {
            let isHidden = hiddenApps.contains(app.name)

            // In normal mode, skip hidden apps
            if !editingAppList && isHidden { continue }

            if editingAppList {
                let row = makeEditableAppRow(name: app.name, tag: index, visible: !isHidden)
                stack.addArrangedSubview(row)
                row.leadingAnchor.constraint(equalTo: stack.leadingAnchor).isActive = true
                row.trailingAnchor.constraint(equalTo: stack.trailingAnchor).isActive = true
            } else {
                let row = makeAppRow(name: app.name, tag: index)
                stack.addArrangedSubview(row)
                row.leadingAnchor.constraint(equalTo: stack.leadingAnchor).isActive = true
                row.trailingAnchor.constraint(equalTo: stack.trailingAnchor).isActive = true
            }
        }

        // Other Coheremote wrapper apps section
        let otherApps = NSWorkspace.shared.runningApplications.filter {
            $0.bundleIdentifier?.hasPrefix("com.coheremote.") == true &&
            $0.processIdentifier != ProcessInfo.processInfo.processIdentifier
        }
        if !otherApps.isEmpty {
            let sepBox = NSBox()
            sepBox.boxType = .separator
            sepBox.translatesAutoresizingMaskIntoConstraints = false
            stack.addArrangedSubview(sepBox)
            sepBox.leadingAnchor.constraint(equalTo: stack.leadingAnchor, constant: 12).isActive = true
            sepBox.trailingAnchor.constraint(equalTo: stack.trailingAnchor, constant: -12).isActive = true

            for app in otherApps {
                let name = app.localizedName ?? "Unknown"
                let row = makeOtherAppRow(name: name, pid: app.processIdentifier, icon: app.icon)
                stack.addArrangedSubview(row)
                row.leadingAnchor.constraint(equalTo: stack.leadingAnchor).isActive = true
                row.trailingAnchor.constraint(equalTo: stack.trailingAnchor).isActive = true
            }
        }
    }

    func makeAppRow(name: String, tag: Int) -> NSView {
        let rowHeight: CGFloat = 30
        let row = NSButton(frame: NSRect(x: 0, y: 0, width: 300, height: rowHeight))
        row.translatesAutoresizingMaskIntoConstraints = false
        row.heightAnchor.constraint(equalToConstant: rowHeight).isActive = true
        row.bezelStyle = .recessed
        row.isBordered = false
        row.title = "  \(name)"
        row.font = NSFont.systemFont(ofSize: 13)

    func makeEditableAppRow(name: String, tag: Int, visible: Bool) -> NSView {
        let rowHeight: CGFloat = 30
        let container = NSView(frame: NSRect(x: 0, y: 0, width: 300, height: rowHeight))
        container.translatesAutoresizingMaskIntoConstraints = false
        container.heightAnchor.constraint(equalToConstant: rowHeight).isActive = true

        // Visibility toggle button
        let toggleBtn = NSButton(frame: NSRect(x: 8, y: 5, width: 20, height: 20))
        toggleBtn.bezelStyle = .recessed
        toggleBtn.isBordered = false
        toggleBtn.tag = tag
        toggleBtn.target = self
        toggleBtn.action = #selector(toggleAppVisibility(_:))
        let symbolName = visible ? "eye" : "eye.slash"
        if let img = NSImage(systemSymbolName: symbolName, accessibilityDescription: nil) {
            img.size = NSSize(width: 14, height: 14)
            toggleBtn.image = img
        }
        toggleBtn.contentTintColor = visible ? .controlAccentColor : .tertiaryLabelColor
        container.addSubview(toggleBtn)

        // App name label
        let nameLabel = NSTextField(labelWithString: name)
        nameLabel.frame = NSRect(x: 34, y: 6, width: 250, height: 18)
        nameLabel.font = NSFont.systemFont(ofSize: 13)
        nameLabel.textColor = visible ? .labelColor : .tertiaryLabelColor
        nameLabel.lineBreakMode = .byTruncatingTail
        container.addSubview(nameLabel)

        return container
    }

    func loadHiddenApps() {
        let key = "Coheremote-\(APP_NAME)-hiddenApps"
        if let arr = UserDefaults.standard.stringArray(forKey: key) {
            hiddenApps = Set(arr)
        }
    }

    func saveHiddenApps() {
        let key = "Coheremote-\(APP_NAME)-hiddenApps"
        UserDefaults.standard.set(Array(hiddenApps), forKey: key)
    }

    @objc func toggleEditMode() {
        editingAppList.toggle()
        editButton?.contentTintColor = editingAppList ? .controlAccentColor : .labelColor
        rebuildAppList()
    }

    @objc func toggleAppVisibility(_ sender: NSButton) {
        let index = sender.tag
        guard index >= 0, index < cachedApps.count else { return }
        let appName = cachedApps[index].name
        if hiddenApps.contains(appName) {
            hiddenApps.remove(appName)
        } else {
            hiddenApps.insert(appName)
        }
        saveHiddenApps()
        rebuildAppList()
    }

    @objc func toggleStartMenu() {
        if startMenuVisible {
            dismissStartMenu()
        } else {
            showStartMenu()
        }
    }

    func showStartMenu() {
        if startMenuPanel == nil { createStartMenuPanel() }
        guard let panel = startMenuPanel else { return }

        updateHeaderStatus()
        rebuildAppList()
        positionStartMenu()
        panel.orderFrontRegardless()
        startMenuVisible = true

        // Click outside to dismiss
        clickMonitor = NSEvent.addGlobalMonitorForEvents(matching: [.leftMouseDown, .rightMouseDown]) { [weak self] _ in
            self?.dismissStartMenu()
        }
    }

    func dismissStartMenu() {
        startMenuPanel?.orderOut(nil)
        startMenuVisible = false
        if editingAppList {
            editingAppList = false
            editButton?.contentTintColor = .labelColor
        }
        if let monitor = clickMonitor {
            NSEvent.removeMonitor(monitor)
            clickMonitor = nil
        }
    }

    func positionStartMenu() {
        guard let panel = startMenuPanel,
              let button = statusItem?.button,
              let btnWindow = button.window else { return }

        let btnRect = button.convert(button.bounds, to: nil)
        let screenRect = btnWindow.convertToScreen(btnRect)

        let panelSize = panel.frame.size
        var x = screenRect.midX - panelSize.width / 2
        let y = screenRect.minY - panelSize.height - 4

        // Keep within screen
        if let screen = btnWindow.screen ?? NSScreen.main {
            let visibleFrame = screen.visibleFrame
            if x + panelSize.width > visibleFrame.maxX {
                x = visibleFrame.maxX - panelSize.width
            }
            if x < visibleFrame.minX {
                x = visibleFrame.minX
            }
        }

        panel.setFrameOrigin(NSPoint(x: x, y: y))
    }

        let menu = NSMenu()

        let restartItem = NSMenuItem(

    @objc func doQuitFromPanel() {
        dismissStartMenu()
        if connected && CONNECTION_MODE != "network" {
            // Show power menu to choose quit method
            let alert = NSAlert()
            alert.messageText = L("Quit \(APP_NAME)", "\(APP_NAME) を終了")
            alert.informativeText = L("Choose how to quit:", "終了方法を選択:")
            alert.addButton(withTitle: L("Suspend VM and Quit", "VMをサスペンドして終了"))
            alert.addButton(withTitle: L("Quit App Only", "アプリだけ終了"))
            alert.addButton(withTitle: L("Cancel", "キャンセル"))
            let response = alert.runModal()
            switch response {
            case .alertFirstButtonReturn:
                doSuspendAndQuit()
            case .alertSecondButtonReturn:
                doQuitAppOnly()
            default:
                break
            }
        } else {
            NSApp.terminate(nil)
        }
    }

    @objc func doLaunchApp(_ sender: Any) {
        let tag: Int
        if let btn = sender as? NSButton { tag = btn.tag }
        else if let item = sender as? NSMenuItem { tag = item.tag }
        else { return }
        guard tag >= 0, tag < cachedApps.count else { return }
        dismissStartMenu()
        launchWindowsApp(path: cachedApps[tag].path)
    }

    @objc func doRefreshApps() {
        cachedApps = []
        loadInstalledApps()
        rebuildAppList()
    }

    @objc func switchToApp(_ sender: Any) {
        let tag: Int
        if let btn = sender as? NSButton { tag = btn.tag }
        else if let item = sender as? NSMenuItem { tag = item.tag }
        else { return }
        dismissStartMenu()
        let pid = Int32(tag)
        if let app = NSWorkspace.shared.runningApplications.first(where: { $0.processIdentifier == pid }) {
            app.activate(options: [])
        }
    }

    @objc func doSuspendAndQuit() {
        dismissStartMenu()
        guard connected, CONNECTION_MODE != "network" else { return }
        connected = false
        showWindow()
        DispatchQueue.global().async {
            self.status(L("Suspending VM...", "VMをサスペンド中..."))
            let (code, output) = self.vmrun("suspend", [VMX_PATH])
            self.log("[VM] vmrun suspend: exit=\(code) output=\(output)")
            self.log("=== Suspend complete, quitting ===")
            DispatchQueue.main.async {
                self.skipCleanup = true
                NSApp.terminate(nil)
            }
        }
    }

    func showWindow() {
        DispatchQueue.main.async {
            self.spinner?.startAnimation(nil)
            self.window?.makeKeyAndOrderFront(nil)
            NSApp.activate(ignoringOtherApps: true)
        }
    }

    func hideWindow() {
        DispatchQueue.main.async {
            self.spinner?.stopAnimation(nil)
            self.window?.orderOut(nil)
        }
    }

    func launchLocalVMMode() {
        ensureFusion()
        startVM()

        status(L("Waiting for VM...", "VMの起動を待機中..."))
        guard waitForVM() else {
            status(L("VM start failed", "VMの起動に失敗"))
            return
        }

        status(L("Connecting...", "接続中..."))
        openRDP()

        Thread.sleep(forTimeInterval: 2)
        hideWindow()
        connected = true

        // Load Windows app list for App Launcher
        if ENABLE_START_MENU && !winPassword.isEmpty {
            loadInstalledApps()
        }
    }

    func launchNetworkMode() {
        // Check if host is already reachable
        if pingHost(HOST_NAME) {
            log("[NET] Host \(HOST_NAME) is already reachable")
        } else {
            // Send WoL if MAC address is configured
            if !MAC_ADDRESS.isEmpty {
                status(L("Sending Wake-on-LAN...", "Wake-on-LANを送信中..."))
                log("[WOL] Sending magic packet to \(MAC_ADDRESS)")
                if sendWoLPacket(mac: MAC_ADDRESS) {
                    log("[WOL] Magic packet sent successfully")
                } else {
                    log("[WOL] Failed to send magic packet")
                }
                Thread.sleep(forTimeInterval: 3)
            }

            // Wait for host to come online
            status(L("Waiting for host...", "ホストの起動を待機中..."))
            guard waitForHost(HOST_NAME, timeout: 120) else {
                status(L("Host unreachable", "ホストに到達できません"))
                log("[ERROR] Host \(HOST_NAME) did not come online within timeout")
                return
            }
            log("[NET] Host \(HOST_NAME) is now reachable")
        }

        // Give the host a moment for RDP service to be ready
        Thread.sleep(forTimeInterval: 5)

        status(L("Connecting...", "接続中..."))
        openRDP()

        Thread.sleep(forTimeInterval: 2)
        hideWindow()
        connected = true
    }

    @objc func doReconnect() {
        dismissStartMenu()
        log("[RDP] Reconnecting...")
        openRDP()
    }

    @objc func doRestart() {
        dismissStartMenu()
        guard connected else { return }
        connected = false
        showWindow()

        DispatchQueue.global().async {
            self.status(L("Restarting Windows...", "Windowsを再起動中..."))
            let (code, output) = self.vmrun("reset", [VMX_PATH, "soft"])
            self.log("[VM] vmrun reset: exit=\(code) output=\(output)")

            if code != 0 {
                self.status(L("Restart command failed", "再起動コマンド失敗"))
                self.log("[ERROR] vmrun reset failed")
                Thread.sleep(forTimeInterval: 3)
                self.hideWindow()
                self.connected = true
                return
            }

            Thread.sleep(forTimeInterval: 15)
            self.status(L("Waiting for Windows...", "Windowsの起動を待機中..."))

            guard self.waitForVM() else {
                self.status(L("Restart timed out", "再起動がタイムアウト"))
                return
            }

            self.status(L("Connecting...", "接続中..."))
            self.openRDP()

            Thread.sleep(forTimeInterval: 2)
            self.hideWindow()
            self.connected = true
        }
    }

    @objc func doShutdownAndQuit() {
        dismissStartMenu()
        guard connected else { return }
        connected = false
        showWindow()

        DispatchQueue.global().async {
            self.status(L("Shutting down Windows...", "Windowsをシャットダウン中..."))
            let (code, output) = self.vmrun("stop", [VMX_PATH, "soft"])
            self.log("[VM] vmrun stop soft: exit=\(code) output=\(output)")

            if code != 0 {
                self.status(L("Shutdown failed, forcing stop...", "シャットダウン失敗、強制停止中..."))
                let (code2, output2) = self.vmrun("stop", [VMX_PATH, "hard"])
                self.log("[VM] vmrun stop hard: exit=\(code2) output=\(output2)")
            }

            self.log("=== Shutdown complete, quitting ===")
            DispatchQueue.main.async {
                NSApp.terminate(nil)
            }
        }
    }

    @objc func doQuitAppOnly() {
        log("[QUIT] Quitting app only (VM will keep running)")
        skipCleanup = true
        NSApp.terminate(nil)
    }

    func applicationShouldTerminate(_ sender: NSApplication) -> NSApplication.TerminateReply {
        unregisterRunning()

        // In network mode, no VM cleanup needed
        if CONNECTION_MODE == "network" {
            log("=== Goodbye ===")
            return .terminateNow
        }

        guard vmIsRunning(), !skipCleanup else {
            log("=== Goodbye ===")
            return .terminateNow
        }

        if otherWrappersUsingVM() {
            log("[CLEANUP] Other wrapper apps are using the same VM, skipping VM cleanup")
            log("=== Goodbye ===")
            return .terminateNow
        }

        if SHUTDOWN_ON_EXIT {
            showWindow()
            DispatchQueue.global().async {
                self.status(L("Shutting down Windows...", "Windowsをシャットダウン中..."))
                let (code, output) = self.vmrun("stop", [VMX_PATH, "soft"])
                self.log("[CLEANUP] vmrun stop soft: exit=\(code) output=\(output)")
                if code != 0 {
                    self.status(L("Shutdown failed, forcing stop...", "シャットダウン失敗、強制停止中..."))
                    let (code2, output2) = self.vmrun("stop", [VMX_PATH, "hard"])
                    self.log("[CLEANUP] vmrun stop hard: exit=\(code2) output=\(output2)")
                }
                self.log("=== Goodbye ===")
                DispatchQueue.main.async {
                    NSApp.reply(toApplicationShouldTerminate: true)
                }
            }
            return .terminateLater
        }

        if CLOSE_ON_EXIT {
            showWindow()
            DispatchQueue.global().async {
                self.status(L("Suspending VM...", "VMをサスペンド中..."))
                let (code, output) = self.vmrun("suspend", [VMX_PATH])
                self.log("[CLEANUP] vmrun suspend: exit=\(code) output=\(output)")
                self.log("=== Goodbye ===")
                DispatchQueue.main.async {
                    NSApp.reply(toApplicationShouldTerminate: true)
                }
            }
            return .terminateLater
        }

        log("=== Goodbye ===")
        return .terminateNow
    }

    func applicationShouldTerminateAfterLastWindowClosed(_ sender: NSApplication) -> Bool {
        return false
    }
